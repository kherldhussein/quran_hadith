# DEB to RPM Package Name Mapping
# Format: debian-package -> rpm-package

# GTK and UI Libraries
libgtk-3-0 -> gtk3
libgtk-3-dev -> gtk3-devel
libglu1-mesa -> mesa-libGLU

# Desktop Integration
libayatana-appindicator3-1 -> libappindicator-gtk3
libayatana-appindicator3-dev -> libappindicator-gtk3-devel
libkeybinder-3.0-0 -> keybinder3
libkeybinder-3.0-dev -> keybinder3-devel

# Compression
liblzma5 -> xz-libs
liblzma-dev -> xz-devel

# Media Playback
libmpv1 -> mpv-libs
libmpv-dev -> mpv-libs-devel

# Audio Codecs
# Note: On Fedora, these require RPM Fusion repository
# Fedora has libavcodec-free (limited codecs) in main repos
libavcodec58 -> libavcodec-free (Fedora), ffmpeg-libs (RPM Fusion)
libavformat58 -> libavformat-free (Fedora), ffmpeg-libs (RPM Fusion)
libavutil56 -> libavutil-free (Fedora), ffmpeg-libs (RPM Fusion)
libswresample3 -> libswresample-free (Fedora), ffmpeg-libs (RPM Fusion)
libswscale5 -> libswscale-free (Fedora), ffmpeg-libs (RPM Fusion)

# Audio Output
libpulse0 -> pulseaudio-libs
libpulse-dev -> pulseaudio-libs-devel
libasound2 -> alsa-lib
libasound2-dev -> alsa-lib-devel

# SSL/TLS
libssl3 -> openssl-libs
libssl-dev -> openssl-devel
ca-certificates -> ca-certificates

# Build Tools
clang -> clang
cmake -> cmake
ninja-build -> ninja-build
pkg-config -> pkgconf-pkg-config
gcc -> gcc
g++ -> gcc-c++

# Notes:
# 1. RPM Fusion: For full codec support on Fedora, users need to enable RPM Fusion
#    sudo dnf install https://download1.rpmfusion.org/free/fedora/rpmfusion-free-release-$(rpm -E %fedora).noarch.rpm
#
# 2. RHEL/CentOS: May need EPEL repository for some packages
#    sudo dnf install epel-release
#
# 3. openSUSE: Most packages available in standard repos
